1. wget https://nodejs.org/dist/v6.3.0/node-v6.3.0.tar.gz
2. tar xzvf node-v* && cd node-v*
3. sudo yum install gcc gcc-c++
4. ./configure
5. make
6. sudo make install
7. node --version *to be sure it installed!!*
